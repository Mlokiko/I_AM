rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Pe³ne profile - mo¿na czytaæ swój profil lub profile swoich opiekunów/podopiecznych
    match /users/{userId} {
      allow read: if request.auth.uid == userId ||
                     get(/databases/$(database)/documents/users/$(request.auth.uid)).data.caregiversID.hasAny([userId]) ||
                     get(/databases/$(database)/documents/users/$(request.auth.uid)).data.caretakersID.hasAny([userId]);
      allow write: if request.auth.uid == userId;
      
      // Zezwól na aktualizacjê caregiversID i caretakersID przez innych u¿ytkowników
      allow update: if request.auth != null &&
                       request.resource.data.keys().hasOnly([
                         'firstName', 'lastName', 'age', 'sex', 'phoneNumber', 
                         'email', 'createdAt', 'isCaregiver', 'caretakersID', 'caregiversID'
                       ]) &&
                       (request.auth.uid == userId || 
                        request.resource.data.diff(resource.data).affectedKeys().hasOnly(['caretakersID', 'caregiversID']));
    }
    
    // Publiczne profile - wszyscy zalogowani mog¹ czytaæ
    match /user_public_profiles/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth.uid == userId;
    }
    
    // Zaproszenia do opiekunów
    match /caregiver_invitations/{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
